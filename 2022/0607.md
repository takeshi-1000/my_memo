## [SwiftUI] Navigation

```
var body: some View {
        NavigationView { // 1.NavigationViewで囲む
            NavigationLink(destination: ContentView()) {　2.NavigationLinkを実装 3.実装の際、遷移したい画面を渡す
                Text("aaa")  // 4.遷移トリガーとなるViewを実装する 
            }
            .navigationTitle("Top View")
        }
    }
```
